        library.save(chooser.getSelectedFile());

    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        ArrayList<String> authors = new ArrayList<>(Arrays.asList(jTextField2.getText().split(",")));
        ArrayList<String> headings = new ArrayList<>(Arrays.asList(jTextField4.getText().split(",")));
       
        Book book = new Book(jTextField1.getText(), authors, Integer.parseInt(jTextField3.getText()), headings, jTextField5.getText(), Integer.parseInt(jTextField6.getText()), jCheckBox1.isSelected());
        library.addBook(book);
        JOptionPane.showMessageDialog(this, "Book has been added to the library");
        
        jTextField1.setText("");
        jTextField2.setText("");
        jTextField3.setText("");
        jTextField4.setText("");
        jTextField5.setText("");
        jTextField6.setText("");
        jCheckBox1.setSelected(false);
    }                                        

    private void jButton2ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        String list = library.SearchBySubject(jTextField7.getText());
        JOptionPane.showMessageDialog(this, "Books details: \n"+list);
    }
